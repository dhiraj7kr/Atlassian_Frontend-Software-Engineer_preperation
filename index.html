JavaScript (ES6+) Examples

1. Variables (let, const, var)

let name = "John";  // Can be reassigned
const age = 30;     // Cannot be reassigned
name = "Jane";      // Valid
// age = 31;         // Error: Assignment to constant variable
console.log(name);  // Output: "Jane"
console.log(age);   // Output: 30

2. Functions (Arrow functions, Regular functions)

function greet(name) {
  return \`Hello, \${name}\`;
}

const greetArrow = (name) => \`Hello, \${name}\`;

console.log(greet("John"));      // "Hello, John"
console.log(greetArrow("Jane")); // "Hello, Jane"

3. Asynchronous JS (Promises, async/await)

function fetchData() {
  return new Promise((resolve) => {
    setTimeout(() => resolve("Data fetched!"), 2000);
  });
}

async function fetchDataAsync() {
  const data = await fetchData();
  console.log(data); // Output: "Data fetched!"
}

fetchDataAsync();

4. Destructuring

const person = { name: "John", age: 30, occupation: "Developer" };

const { name, age } = person;

console.log(\`Name: \${name}, Age: \${age}\`);  // Output: "Name: John, Age: 30"
